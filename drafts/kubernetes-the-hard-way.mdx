## Local VM Setup

```sh
brew install lima

for name in jumpbox server node-0 node-1; do
  # create the VM
  limactl start \
    --name "$name" \
    --arch=aarch64 \
    --cpus=2 \
    --memory=2 \
    --disk=20 \
    template:debian-12

    # set the VMs hostname
    limactl shell "$name" -- sudo hostnamectl set-hostname "$name"
done
```

for name in jumpbox server node-0 node-1; do
  echo "$name:"
